    Ubuntu 使用教程

    更新时间：2026年5月 适用环境：Ubuntu 24.04 LTS · 1H-ARCH 自用部署（ub-server 192.168.31.84）

    一、系统概述

    Ubuntu 是基于 Debian 的 Linux 发行版，由 Canonical 公司维护，是最流行的桌面与服务器 Linux 发行版之一。LTS（长期支持）版本提供 5 年安全更新。

    本部署环境：

    项目 配置
    主机名 ub-server
    内网 IP 192.168.31.84
    ZeroTier IP 10.166.37.64
    系统 Ubuntu 24.04.4 LTS (Noble Numbat)
    SSH 用户 harry
    用途 Hermes Agent 主机、SMB 文件共享、Wiki 服务

    二、系统安装

    2.1 下载镜像

    # 官方下载（桌面版 / 服务器版）
https://ubuntu.com/download

# 国内镜像
https://mirrors.tuna.tsinghua.edu.cn/ubuntu-releases/24.04/
https://mirrors.aliyun.com/ubuntu-releases/24.04/

    2.2 制作启动盘

    # macOS
brew install --cask balenaetcher
# 或使用 dd
diskutil list
diskutil unmountDisk /dev/diskN
sudo dd if=ubuntu-24.04-live-server-amd64.iso of=/dev/diskN bs=1m

# Windows
# 使用 Rufus: https://rufus.ie
# 选择 ISO → GPT → UEFI → 写入

    2.3 安装步骤

    1. 插入 U 盘，BIOS 设置 UEFI 启动
    2. 选择语言（建议 English，后续可装中文包）
    3. 网络配置（DHCP 或静态 IP）
    4. 磁盘分区（自动 / 手动）
    5. 创建用户和密码
    6. 选择是否安装 OpenSSH Server（推荐勾选）
    7. 等待安装完成，重启

    2.4 首次登录

    # SSH 远程登录
ssh harry@192.168.31.84

# 本地控制台登录后查看 IP
ip addr show

    三、基础配置

    3.1 更换国内镜像源

    # 备份原始源
sudo cp /etc/apt/sources.list.d/ubuntu.sources /etc/apt/sources.list.d/ubuntu.sources.bak

# 替换为清华源（Ubuntu 24.04 新格式）
sudo tee /etc/apt/sources.list.d/ubuntu.sources << 'EOF'
Types: deb
URIs: https://mirrors.tuna.tsinghua.edu.cn/ubuntu
Suites: noble noble-updates noble-backports
Components: main restricted universe multiverse
Signed-By: /usr/share/keyrings/ubuntu-archive-keyring.gpg

Types: deb
URIs: https://mirrors.tuna.tsinghua.edu.cn/ubuntu
Suites: noble-security
Components: main restricted universe multiverse
Signed-By: /usr/share/keyrings/ubuntu-archive-keyring.gpg
EOF

# 更新索引
sudo apt update && sudo apt upgrade -y

    3.2 设置静态 IP

    # Ubuntu 24.04 使用 Netplan
sudo nano /etc/netplan/00-installer-config.yaml

    network:
  version: 2
  ethernets:
    eth0:
      dhcp4: no
      addresses:
        - 192.168.31.84/24
      routes:
        - to: default
          via: 192.168.31.1
      nameservers:
        addresses:
          - 223.5.5.5
          - 114.114.114.114

    # 应用配置
sudo netplan apply

# 验证
ip addr show eth0
ping -c 3 223.5.5.5

    3.3 时区与时间同步

    # 设置时区为上海
sudo timedatectl set-timezone Asia/Shanghai

# 启用 NTP 同步
sudo timedatectl set-ntp true

# 验证
timedatectl

    3.4 创建用户与 sudo 权限

    # 创建新用户
sudo adduser newuser

# 加入 sudo 组
sudo usermod -aG sudo newuser

# 切换用户
su - newuser

# 验证 sudo 权限
sudo whoami

    四、SSH 服务

    4.1 安装与配置

    # 安装（通常安装时已勾选）
sudo apt install openssh-server -y

# 查看状态
sudo systemctl status ssh

# 启动 / 停止 / 重启
sudo systemctl start ssh
sudo systemctl stop ssh
sudo systemctl restart ssh

# 开机自启
sudo systemctl enable ssh

    4.2 SSH 密钥登录（推荐）

    # 本地生成密钥对
ssh-keygen -t ed25519 -C "harry@1h-arch"

# 复制公钥到服务器
ssh-copy-id harry@192.168.31.84

# 或手动复制
mkdir -p ~/.ssh
chmod 700 ~/.ssh
echo "ssh-ed25519 AAAA... harry@1h-arch" >> ~/.ssh/authorized_keys
chmod 600 ~/.ssh/authorized_keys

    4.3 加固 SSH 配置

    sudo nano /etc/ssh/sshd_config

    # 修改以下配置
Port 22                          # 可改为非标准端口
PermitRootLogin no               # 禁止 root 登录
PasswordAuthentication no        # 禁用密码登录（启用密钥后）
MaxAuthTries 3                   # 最大尝试次数
ClientAliveInterval 300          # 保持连接
ClientAliveCountMax 2

    # 重启生效
sudo systemctl restart ssh

    4.4 SSH 常用技巧

    # 保持连接不断开（客户端配置 ~/.ssh/config）
Host *
    ServerAliveInterval 60
    ServerAliveCountMax 3

# 端口转发（本地 → 远程）
ssh -L 8080:localhost:80 harry@192.168.31.84

# 端口转发（远程 → 本地）
ssh -R 9090:localhost:3000 harry@192.168.31.84

# SSH 跳板机
ssh -J jumpuser@jumphost harry@192.168.31.84

# 传输文件
scp file.txt harry@192.168.31.84:/home/harry/
scp harry@192.168.31.84:/home/harry/file.txt ./

    五、文件共享（SMB）

    5.1 安装 Samba

    sudo apt install samba samba-common-bin -y

# 查看版本
smbd --version

    5.2 配置共享目录

    # 创建共享目录
sudo mkdir -p /srv/samba/mywiki
sudo chown harry:harry /srv/samba/mywiki

# 编辑配置
sudo nano /etc/samba/smb.conf

    [mywiki]
   path = /srv/samba/mywiki
   browseable = yes
   read only = no
   valid users = harry
   create mask = 0644
   directory mask = 0755
   veto files = /.DS_Store/

    5.3 设置 Samba 密码

    # 为用户设置 SMB 密码（独立于系统密码）
sudo smbpasswd -a harry

# 启用用户
sudo smbpasswd -e harry

    5.4 启动与验证

    # 重启服务
sudo systemctl restart smbd nmbd

# 查看状态
sudo systemctl status smbd

# 本地测试
smbclient -U harry //192.168.31.84/mywiki

# 列出所有共享
smbclient -L //192.168.31.84 -U harry

    5.5 各平台访问

    # Windows
\\192.168.31.84\mywiki

# macOS Finder
smb://192.168.31.84/mywiki

# macOS 命令行
mkdir -p /tmp/smb_wiki
mount_smbfs //harry@192.168.31.84/mywiki /tmp/smb_wiki

# Linux
sudo mount -t cifs //192.168.31.84/mywiki /mnt/wiki -o user=harry

    六、软件包管理

    6.1 APT 基础命令

    # 更新软件源索引
sudo apt update

# 升级所有已安装软件
sudo apt upgrade -y

# 完整升级（可处理依赖变更）
sudo apt full-upgrade -y

# 安装软件
sudo apt install <package>

# 卸载软件（保留配置）
sudo apt remove <package>

# 彻底卸载（含配置）
sudo apt purge <package>

# 清理缓存
sudo apt autoclean
sudo apt autoremove -y

# 搜索软件
apt search <keyword>

# 查看软件信息
apt show <package>

# 列出已安装软件
apt list --installed

    6.2 Snap 包管理

    # 查看已安装
snap list

# 安装
sudo snap install <package>

# 更新
sudo snap refresh <package>

# 卸载
sudo snap remove <package>

    6.3 常用软件安装

    # 开发工具
sudo apt install -y build-essential git curl wget vim nano htop

# Python 环境
sudo apt install -y python3 python3-pip python3-venv

# Node.js（通过 NodeSource）
curl -fsSL https://deb.nodesource.com/setup_22.x | sudo -E bash -
sudo apt install -y nodejs

# Docker（官方脚本）
curl -fsSL https://get.docker.com | sudo sh
sudo usermod -aG docker $USER

# 网络工具
sudo apt install -y net-tools dnsutils traceroute nmap

# 压缩解压
sudo apt install -y zip unzip p7zip-full

    七、Docker 容器服务

    7.1 安装 Docker

    # 官方安装脚本
curl -fsSL https://get.docker.com | sudo sh

# 加入 docker 组（免 sudo）
sudo usermod -aG docker $USER
newgrp docker

# 验证
docker --version
docker run hello-world

    7.2 配置国内镜像加速

    sudo tee /etc/docker/daemon.json << 'EOF'
{
  "registry-mirrors": [
    "https://docker.mirrors.ustc.edu.cn",
    "https://hub-mirror.c.163.com"
  ]
}
EOF

sudo systemctl daemon-reload
sudo systemctl restart docker

    7.3 常用 Docker 命令

    # 镜像管理
docker images                          # 列出镜像
docker pull <image>                    # 拉取镜像
docker rmi <image>                     # 删除镜像

# 容器管理
docker ps                              # 运行中的容器
docker ps -a                           # 所有容器
docker run -d --name myapp -p 8080:80 <image>   # 后台运行
docker stop <container>                # 停止
docker start <container>               # 启动
docker restart <container>             # 重启
docker rm <container>                  # 删除
docker logs <container>                # 查看日志
docker logs -f <container>             # 跟踪日志

# 进入容器
docker exec -it <container> /bin/bash

# 容器资源
docker stats                           # 实时资源占用
docker system df                       # 磁盘占用
docker system prune -a                 # 清理无用资源

    7.4 Docker Compose

    # 安装
sudo apt install docker-compose-plugin -y

# 验证
docker compose version

    # docker-compose.yml 示例
services:
  web:
    image: nginx:latest
    ports:
      - "80:80"
    volumes:
      - ./html:/usr/share/nginx/html
    restart: unless-stopped

  db:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: secret
    volumes:
      - db_data:/var/lib/mysql
    restart: unless-stopped

volumes:
  db_data:

    # 启动
docker compose up -d

# 停止
docker compose down

# 查看日志
docker compose logs -f

    八、Systemd 服务管理

    8.1 基础命令

    # 服务状态
systemctl status <service>

# 启动 / 停止 / 重启
sudo systemctl start <service>
sudo systemctl stop <service>
sudo systemctl restart <service>

# 开机自启
sudo systemctl enable <service>
sudo systemctl disable <service>

# 查看所有服务
systemctl list-units --type=service
systemctl list-units --type=service --state=running

# 查看日志
journalctl -u <service> -f            # 跟踪日志
journalctl -u <service> --since today  # 今天的日志
journalctl -u <service> -n 50          # 最近50行

    8.2 自定义服务示例

    # /etc/systemd/system/myapp.service
[Unit]
Description=My Application
After=network.target

[Service]
Type=simple
User=harry
WorkingDirectory=/home/harry/myapp
ExecStart=/home/harry/myapp/start.sh
Restart=always
RestartSec=10
Environment=NODE_ENV=production

[Install]
WantedBy=multi-user.target

    # 部署服务
sudo systemctl daemon-reload
sudo systemctl enable myapp
sudo systemctl start myapp
sudo systemctl status myapp

    九、防火墙（UFW）

    9.1 基础配置

    # 安装
sudo apt install ufw -y

# 默认策略
sudo ufw default deny incoming
sudo ufw default allow outgoing

# 开放端口
sudo ufw allow 22/tcp          # SSH
sudo ufw allow 80/tcp          # HTTP
sudo ufw allow 443/tcp         # HTTPS
sudo ufw allow 445/tcp         # SMB
sudo ufw allow from 192.168.31.0/24  # 内网全放行

# 启用
sudo ufw enable

# 查看状态
sudo ufw status verbose

    9.2 常用规则

    # 开放特定 IP 访问
sudo ufw allow from 192.168.31.187 to any port 22

# 开放端口范围
sudo ufw allow 6000:6010/tcp

# 删除规则
sudo ufw delete allow 80/tcp

# 重置所有规则
sudo ufw reset

    十、定时任务

    10.1 Cron 定时任务

    # 编辑当前用户的 crontab
crontab -e

# 查看当前任务
crontab -l

    # 格式：分 时 日 月 周 命令
# ┌──────── 分钟 (0-59)
# │ ┌────── 小时 (0-23)
# │ │ ┌──── 日 (1-31)
# │ │ │ ┌── 月 (1-12)
# │ │ │ │ ┌ 星期 (0-7, 0和7都是周日)
# │ │ │ │ │
# * * * * * command

# 示例
0 2 * * *     /home/harry/backup.sh          # 每天凌晨2点
0 */6 * * *   /home/harry/sync.sh            # 每6小时
30 8 * * 1-5  /home/harry/workday.sh         # 工作日8:30
0 0 1 * *     /home/harry/monthly.sh         # 每月1号
*/10 * * * *  /home/harry/check.sh           # 每10分钟

    10.2 Systemd Timer（推荐替代方案）

    # /etc/systemd/system/backup.timer
[Unit]
Description=Daily Backup Timer

[Timer]
OnCalendar=*-*-* 02:00:00
Persistent=true

[Install]
WantedBy=timers.target

    # /etc/systemd/system/backup.service
[Unit]
Description=Daily Backup

[Service]
Type=oneshot
User=harry
ExecStart=/home/harry/backup.sh

    sudo systemctl daemon-reload
sudo systemctl enable backup.timer
sudo systemctl start backup.timer

# 查看所有 timer
systemctl list-timers

    十一、磁盘与文件系统

    11.1 磁盘管理

    # 查看磁盘
lsblk
sudo fdisk -l

# 查看使用情况
df -h
du -sh /path/to/dir         # 目录大小
du -h --max-depth=1 /home   # 逐层大小

# 挂载磁盘
sudo mkdir -p /mnt/data
sudo mount /dev/sdb1 /mnt/data

# 开机自动挂载（/etc/fstab）
/dev/sdb1  /mnt/data  ext4  defaults  0  2
# 或用 UUID（更稳定）
UUID=xxxx-xxxx  /mnt/data  ext4  defaults  0  2

# 查看 UUID
blkid /dev/sdb1

    11.2 文件权限

    # 基本权限
chmod 755 script.sh          # rwxr-xr-x
chmod 644 file.txt           # rw-r--r--
chmod +x script.sh           # 添加执行权限

# 递归修改
chmod -R 755 /path/to/dir
chown -R harry:harry /path/to/dir

# 特殊权限
chmod +s /usr/bin/somebin    # SUID
chmod +t /tmp                # Sticky bit

    11.3 软链接与硬链接

    # 软链接（快捷方式，可跨分区）
ln -s /data/photos ~/photos_link

# 硬链接（同一文件，不能跨分区）
ln /data/file.txt /backup/file.txt

# 查看链接指向
readlink -f ~/photos_link

    十四、进程与性能监控

    # 内存

    十五、日志与排障

    15.1 系统日志

    # systemd 日志
journalctl -xe                          # 最新错误
journalctl -u ssh -f                    # SSH 服务日志
journalctl --since "1 hour ago"         # 最近1小时
journalctl -p err                       # 仅错误级别

# 传统日志文件
tail -f /var/log/syslog
tail -f /var/log/auth.log               # 登录日志
tail -f /var/log/kern.log               # 内核日志

    15.2 常见问题排查

    SSH 连接被拒： 

    sudo systemctl status sshd
sudo ufw status
sudo journalctl -u ssh --since "10 min ago"

    磁盘满： 

    df -h
du -h --max-depth=1 / | sort -hr | head -20
sudo find / -xdev -type f -size +500M 2>/dev/null
sudo journalctl --vacuum-size=100M      # 清理日志

    内存不足： 

    free -h
ps aux --sort=-%mem | head -10
sudo systemctl restart <memory-hungry-service>

    端口被占用： 

    sudo ss -tlnp | grep :8080

    十六、备份与恢复

    16.1 rsync 同步

    # 本地同步
rsync -avz /source/ /destination/

# 远程同步（推）
rsync -avz -e ssh /local/dir/ harry@192.168.31.84:/remote/dir/

# 远程同步（拉）
rsync -avz -e ssh harry@192.168.31.84:/remote/dir/ /local/dir/

# 删除目标多余文件（镜像同步）
rsync -avz --delete /source/ /destination/

# 排除文件
rsync -avz --exclude='*.log' --exclude='node_modules' /source/ /dest/

    16.2 tar 归档

    # 压缩
tar czf backup_$(date +%Y%m%d).tar.gz /home/harry/important/

# 解压
tar xzf backup_20260508.tar.gz

# 查看内容
tar tzf backup_20260508.tar.gz

    16.3 定时备份脚本

    #!/bin/bash
# /home/harry/backup.sh
BACKUP_DIR="/home/harry/backups"
SOURCE_DIR="/home/harry/important"
REMOTE_HOST="harry@192.168.31.187"
REMOTE_DIR="/vol1/ub_backups"
DATE=$(date +%Y%m%d_%H%M)

# 本地备份
tar czf "${BACKUP_DIR}/backup_${DATE}.tar.gz" "$SOURCE_DIR"

# 远程同步
rsync -avz "${BACKUP_DIR}/" "${REMOTE_HOST}:${REMOTE_DIR}/"

# 清理30天前的备份
find "${BACKUP_DIR}" -name "backup_*.tar.gz" -mtime +30 -delete

echo "Backup completed: ${DATE}"

    17.2 Fail2Ban（防暴力破解）

    # 安装
sudo apt install fail2ban -y

# 配置
sudo tee /etc/fail2ban/jail.local << 'EOF'
[DEFAULT]
bantime = 3600
findtime = 600
maxretry = 3

[sshd]
enabled = true
port = ssh
filter = sshd
logpath = /var/log/auth.log
maxretry = 3
EOF

# 启动
sudo systemctl enable fail2ban
sudo systemctl start fail2ban

# 查看封禁状态
sudo fail2ban-client status sshd
